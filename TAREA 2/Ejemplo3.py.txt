def celsius_a_fahrenheit(temp_c): #donde "temp_c" temperatura en grados celcius y "temp_f", en grados fahrenheit
    '''
    transforma temperatura de celsius a fahrenheit.
    Input: 
        temp_c:número float
    Ouput: 
        retorna: número float
    '''
    
    temp_f = 1.8*temp_c + 32
    
    return temp_f
